    // === Timer pentru Bac ===
    const bacDate = new Date("June 22, 2026 09:00:00").getTime();
    const timer = setInterval(function() {
      const now = new Date().getTime();
      const distance = bacDate - now;

      const days = Math.floor(distance / (1000 * 60 * 60 * 24));
      const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
      const seconds = Math.floor((distance % (1000 * 60)) / 1000);

      document.getElementById("timer").innerHTML =
        days + "z " + hours + "h " + minutes + "m " + seconds + "s ";

      if (distance < 0) {
        clearInterval(timer);
        document.getElementById("timer").innerHTML = "Succes la Bac! üéâ";
      }
    }, 1000);

    // === AdƒÉugare eseuri (doar local, nu salvate pe net) ===
    function adaugaEseu() {
      const input = document.getElementById("eseuInput");
      const text = input.value.trim();

      if (text !== "") {
        const li = document.createElement("li");
        li.textContent = text;
        document.getElementById("listaEseuri").appendChild(li);
        input.value = "";
      }
    }
